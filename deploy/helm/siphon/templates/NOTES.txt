{{- if not .Values.api.auth.secretName }}

⚠ siphon-api is running WITHOUT a bearer-token secret. Machine
  clients won't be able to authenticate through nginx. Create the
  secret and set `api.auth.secretName`:

    kubectl -n {{ .Release.Namespace }} create secret generic siphon-api-auth \
      --from-literal=api-key="$(openssl rand -hex 32)"

    helm upgrade {{ .Release.Name }} ./deploy/helm/siphon \
      --set api.auth.secretName=siphon-api-auth
{{- end }}

{{- if and .Values.authelia.enabled (not .Values.authelia.secretName) }}

⚠ Authelia is using placeholder JWT / session / storage seeds.
  Create the secret and set `authelia.secretName`:

    kubectl -n {{ .Release.Namespace }} create secret generic siphon-authelia \
      --from-literal=jwt_secret="$(openssl rand -hex 32)" \
      --from-literal=session_secret="$(openssl rand -hex 32)" \
      --from-literal=storage_encryption_key="$(openssl rand -hex 32)"
{{- end }}
